1. Обработчик клика и верстка в ModalProject.jsx
Проверка на клик по фону: Убедились, что клик по фону модального окна (класс modal-project-container) вызывает функцию closeModal.
import React from 'react';
import './modalProject.css';

export default function ModalProject({ modalActive, closeModal, name }) {
    const handleClick = (e) => {
        if (e.target.classList.contains('modal-project-container')) {
            closeModal();
        }
    };

    return (
        <div
            className={`modal-project-container ${modalActive ? 'active' : ''}`}
            onClick={handleClick}
        >
            <div className="modal-project-content">
                {name}
                <p>
                    <a onClick={closeModal} style={{ cursor: 'pointer' }}>
                        close
                    </a>
                </p>
            </div>
        </div>
    );
}

3. Компонент Project.jsx
Корректное обновление состояния: Убедились, что состояние modalActive корректно обновляется и используется для управления видимостью модального окна.
Правильная структура JSX: Убедились, что обработчик клика openModal привязан к нужным элементам.
import React, { useState } from 'react';
import ModalProject from './ModalProject';
import './project.css';

export default function Project({ projectData }) {
    const { name, description, technologies, source, site } = projectData;

    const [modalActive, setModalActive] = useState(false);

    const openModal = () => {
        setModalActive(true);
    };

    const closeModal = () => {
        console.log('Closing modal');
        setModalActive(false);
    };

    return (
        <div className="project-container">
            <div className="project-main" onClick={openModal}>
                <p>{name}</p>
                <p>{description}</p>
                <p>{technologies ? technologies.join(', ') : 'Will be soon'}</p>
            </div>
            <div className="project-buttons">
                <button className="project-button">
                    <a href={source}>Source code</a>
                </button>
                <button className="project-button">
                    <a href={site}>Demo site</a>
                </button>
                <button className="project-button" onClick={openModal}>See case</button>
            </div>
            <ModalProject
                modalActive={modalActive}
                closeModal={closeModal}
                name={name}
            />
        </div>
    );
}

Название проекта: TaskManager Pro

Краткое описание: Платформа для управления задачами и проектами, ориентированная на малые и средние команды. Включает в себя календарь, систему уведомлений и интеграцию с популярными инструментами (Trello, Slack).

Роль: Ведущий разработчик

Основные функции:

Создание и управление задачами.
Группировка задач по проектам и меткам.
Интеграция с календарем Google и Slack.
Уведомления и напоминания.
Технологии:

Backend: Node.js, Express.js
Frontend: React, Redux
База данных: MongoDB
Хостинг: AWS
Вызовы и решения: Основная сложность заключалась в создании системы уведомлений в реальном времени. Для этого был использован WebSocket с серверной поддержкой на Node.js, что позволило обеспечить мгновенные обновления и высокую производительность.

Результаты: С момента запуска платформы было зарегистрировано более 500 команд, суммарное количество созданных задач превысило 10 000.
